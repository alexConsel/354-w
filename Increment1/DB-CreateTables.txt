CREATE TABLE users (user_id TEXT PRIMARY KEY NOT NULL, user_pwd TEXT NOT NULL, f_name TEXT, l_name TEXT);

CREATE TABLE projects (project_id INTEGER PRIMARY KEY AUTOINCREMENT, project_name TEXT, owner_id TEXT, is_done INTEGER CHECK(0 OR 1), FOREIGN KEY(owner_id) REFERENCES users(user_id));

CREATE TABLE tasks (task_id INTEGER PRIMARY KEY AUTOINCREMENT, task_name TEXT, description TEXT, duration INTEGER NOT NULL, project_id INTEGER, user_id TEXT, FOREIGN KEY(project_id) REFERENCES projects(project_id), FOREIGN KEY(user_id) REFERENCES users(user_id));

CREATE TABLE precedence (task_id INTEGER, pre_req INTEGER, FOREIGN KEY(task_id) REFERENCES tasks(task_id), FOREIGN KEY(task_id) REFERENCES tasks(task_id));
